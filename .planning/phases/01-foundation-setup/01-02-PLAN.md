       ```typescript
       import { Pool } from 'pg';
       import { validateEnv } from '../config/env.js';

       const env = validateEnv();

       export const pool = new Pool({
         host: env.DB_HOST,
         port: env.DB_PORT,
         database: env.DB_NAME,
         user: env.DB_USER,
         password: env.DB_PASSWORD,
         max: 20,
         idleTimeoutMillis: 30000,
         connectionTimeoutMillis: 2000,
       });

       pool.on('error', (err) => {
         console.error('Unexpected PostgreSQL pool error:', err);
       });

       export async function closePool(): Promise<void> {
         await pool.end();
       }

       // Health check function
       export async function checkConnection(): Promise<boolean> {
         try {
           const client = await pool.connect();
           await client.query('SELECT 1');
           client.release();
           return true;
         } catch (error) {
           console.error('Database connection failed:', error);
           return false;
         }
       }
       ```

    2. Create src/db/migrations/0001-initial-schema.sql:
       ```sql
       -- Locations table for Singapore dog runs
       CREATE TABLE IF NOT EXISTS locations (
         id SERIAL PRIMARY KEY,
         name VARCHAR(255) NOT NULL UNIQUE,
         region VARCHAR(100) NOT NULL,
         latitude DECIMAL(10, 8) NOT NULL,
         longitude DECIMAL(11, 8) NOT NULL,
         notes TEXT,
         created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
       );

       -- Index for region-based queries
       CREATE INDEX IF NOT EXISTS idx_locations_region ON locations(region);

       -- Index for coordinate-based lookups (geofencing)
       CREATE INDEX IF NOT EXISTS idx_locations_coordinates ON locations(latitude, longitude);
       ```

    3. Create src/db/migrate.ts to run migrations:
       ```typescript
       import { pool, closePool } from './client.js';
       import { readFileSync } from 'fs';
       import { join, dirname } from 'path';
       import { fileURLToPath } from 'url';

       const __dirname = dirname(fileURLToPath(import.meta.url));

       async function migrate(): Promise<void> {
         console.log('Running database migrations...');

         try {
           // Read and execute migration file
           const migrationPath = join(__dirname, 'migrations', '0001-initial-schema.sql');
           const migrationSQL = readFileSync(migrationPath, 'utf-8');

           await pool.query(migrationSQL);
           console.log('Migration 0001-initial-schema.sql applied successfully');

         } catch (error) {
           console.error('Migration failed:', error);
           process.exit(1);
         } finally {
           await closePool();
         }
       }

       migrate();
       ```

    IMPORTANT:
    - Use connection pooling (max: 20) to handle concurrent requests efficiently
    - Include error event handler on pool to catch unexpected errors
    - Export checkConnection() for health checks in bot startup
    - Use IF NOT EXISTS to make migrations idempotent

       ```typescript
       import { pool } from './client.js';

       export interface Location {
         id: number;
         name: string;
         region: string;
         latitude: number;
         longitude: number;
         notes: string | null;
         created_at: Date;
       }

       /**
        * Get all locations, optionally filtered by region
        */
       export async function getAllLocations(region?: string): Promise<Location[]> {
         if (region) {
           const result = await pool.query<Location>(
             'SELECT * FROM locations WHERE region = $1 ORDER BY name',
             [region]
           );
           return result.rows;
         }

         const result = await pool.query<Location>(
           'SELECT * FROM locations ORDER BY name'
         );
         return result.rows;
       }

       /**
        * Get a single location by ID
        */
       export async function getLocationById(id: number): Promise<Location | null> {
         const result = await pool.query<Location>(
           'SELECT * FROM locations WHERE id = $1',
           [id]
         );
         return result.rows[0] || null;
       }

       /**
        * Get a single location by name
        */
       export async function getLocationByName(name: string): Promise<Location | null> {
         const result = await pool.query<Location>(
           'SELECT * FROM locations WHERE name = $1',
           [name]
         );
         return result.rows[0] || null;
       }
       ```
